def f(x):
    return np.sin(x)

def df(x):
    return np.cos(x)

def get_df_approx(alpha, h, x_s):
    x0 = x_s - h / alpha
    y0 = f(x0)
    x1 = x_s
    y1 = f(x1)
    x2 = x_s + h * alpha
    y2 = f(x2)
    def df_approx(x):
        return 1 / h * ( (y1 - y0) * (alpha - 1 / alpha) + (y2 - y0) / (alpha ** 3 + alpha) )
    return df_approx    

def draw_error(alpha, x_s):
    data_x = []
    data_y = []
    step = 0.001
    h_min = 10 ** (-8)
    h_max = 1
    h = h_min
    real_value = df(x_s)
    while h <= h_max:
        approx_value = get_df_approx(alpha, h, x_s)(x_s)
        data_x.append(h)
        data_y.append(abs(approx_value - real_value))
        h += step
    plt.subplot(211)
    plt.title("Error for x_s = " + str(x_s))
    plt.plot(data_x, data_y)
    plt.ylabel("|df_real(x_s) - df_approx(x_s)|")
    plt.xlabel("h")

def calculate_trapeziod(f, a, b, M):
    res = 0
    H = (b - a) / M
    x_cur = a + H
    for i in range(1, M):
        res += f(x_cur)
        x_cur += H
    res += f(a) / 2
    res += f(b) / 2
    res *= H
    return res

def calculate_Simpson(f, a, b, M):
    H = (b - a) / M
    res = f(a) + f(b) + 4 * f(a + H / 2)
    x_cur = a + H
    for i in range(1, M):
        res += 2 * f(x_cur) + 4 * f(x_cur + H / 2)
        x_cur += H
    res *= H / 6
    return res

def get_weights(a, b, N, calculation_method, M):
    def f_to_integrate(i):
        def f(q):
            prod = 1
            for k in range(1, N + 1):
                if k != i:
                    prod *= q - (k - 1)
            return prod
        return f
    
    h = (b - a) / (N - 1)
    lambdas = []
    for i in range(1, N + 1):
        l = calculation_method(f_to_integrate(i), 0, N - 1, M)
        l *= (-1) ** (N - i) * h / np.math.factorial(i - 1) / np.math.factorial(N - i) 
        lambdas.append(l)
    return lambdas

def get_first_N_when_negative(a, b, calculation_method, M):
    N = 2
    min_weight = min(get_weights(a, b, N, calculation_method, M))
